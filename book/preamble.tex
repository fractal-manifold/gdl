% ----- Robust \includegraphics --------------------------------------
% If a referenced image file cannot be found (e.g. the images/ folder
% has not been added yet) draw a labelled placeholder so the document
% still compiles. When the real file is present it is used as normal.
\makeatletter
\let\gdl@origincludegraphics\includegraphics
\newif\ifgdl@imgfound
\def\gdl@extlist{,.pdf,.png,.jpg,.jpeg,.PDF,.PNG,.eps}
\newcommand{\gdl@testimage}[1]{%
  \gdl@imgfoundfalse
  \@for\gdl@ext:=\gdl@extlist\do{%
    \IfFileExists{#1\gdl@ext}{\gdl@imgfoundtrue}{}%
    \IfFileExists{images/#1\gdl@ext}{\gdl@imgfoundtrue}{}%
    \IfFileExists{figures/#1\gdl@ext}{\gdl@imgfoundtrue}{}%
  }%
}
\renewcommand{\includegraphics}[2][]{%
  \gdl@testimage{#2}%
  \ifgdl@imgfound
    \gdl@origincludegraphics[#1]{#2}%
  \else
    \fbox{\begin{minipage}{0.6\linewidth}\centering
      \ttfamily\scriptsize [imagen pendiente]\\\detokenize{#2}\end{minipage}}%
  \fi
}
\makeatother

% ----- Spanish / TikZ shorthand clash -------------------------------
% babel-spanish makes < and > active (for << >> guillemets), which breaks
% TikZ arrow syntax such as ->, >=stealth and <-. Disable just those two
% shorthands for the Spanish edition; accents are written explicitly (\'o).
% babel activates its shorthands at \begin{document}, so disabling them must
% happen there too (a preamble \shorthandoff would be undone).
\def\esbooklang{es}
\ifx\booklang\esbooklang
  \AtBeginDocument{\shorthandoff{<>}}
\fi
