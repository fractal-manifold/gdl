% ----- Spanish / TikZ shorthand clash -------------------------------
% babel-spanish makes < and > active (for << >> guillemets), which breaks
% TikZ arrow syntax such as ->, >=stealth and <-. Disable just those two
% shorthands for the Spanish edition; accents are written explicitly (\'o).
% babel activates its shorthands at \begin{document}, so disabling them must
% happen there too (a preamble \shorthandoff would be undone).
\def\esbooklang{es}
\ifx\booklang\esbooklang
  \AtBeginDocument{\shorthandoff{<>}}
\fi
