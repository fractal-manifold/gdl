% =====================================================================
%  Amazon KDP full wraparound cover (standalone, one page).
%  Layout left -> right: [ back cover | spine | front cover ], full bleed.
%  Do NOT compile by hand: the Makefile generates cover-dimensions.tex
%  (paper size, spine width, language, spine-text flag) from the print
%  interior's page count, derives figures/cover-wrap.jpg, then runs
%  `latexmk cover.tex` and renames the result to cover-<lang>.pdf.
%
%  Dimension contract (KDP, white paper), all from cover-dimensions.tex:
%    \bleed       0.125in   (with unit)        \Bn  0.125   (unitless)
%    \spinewidth  pages*0.002252 in (w/ unit)  \Sn          (unitless)
%    \coverW = 0.25 + 14 + spine  in           \coverH = 10.25in
%    \coverlang   en | es        \ifspinetext  (gated on page count)
%  Panel x-edges (inches from the left, no bleed BETWEEN panels):
%    back  [\Bn, \Bn+7]   spine [\Bn+7, \Bn+7+\Sn]   front [\Bn+7+\Sn, \Bn+14+\Sn]
%  Sheet vertical centre is y = 5.125 (half of 10.25in).
% =====================================================================
\documentclass{article}
\newif\ifspinetext
\input{cover-dimensions}          % generated by the Makefile
\input{lang/\coverlang}           % reuse the localized \book* macros

% Compiled with lualatex (see the Makefile) so the cover can use the
% Bruno Ace display font, vendored in fonts/ and loaded by file path so
% the build is reproducible (incl. CI, which lacks the system font).
\usepackage{fontspec}
\setmainfont{Latin Modern Roman}                       % readable back-cover blurb
\newfontface\coverfont{BrunoAce-Regular.ttf}[Path=fonts/]  % title / spine display
\usepackage[\babellang]{babel}
\usepackage[paperwidth=\coverW,paperheight=\coverH,margin=0pt]{geometry}
\usepackage{graphicx}
\graphicspath{{figures/}}
\usepackage{tikz}
\usetikzlibrary{calc}
\usepackage{xcolor}
\usepackage[outline]{contour}  % outlined single-line display text

\pagestyle{empty}
\setlength{\parindent}{0pt}

\begin{document}%
\begin{tikzpicture}[remember picture, overlay, x=1in, y=1in]
  \coordinate (O) at (current page.south west);

  % --- Full-bleed mirrored background across the whole sheet ----------
  \node[anchor=south west, inner sep=0] at (O)
    {\includegraphics[width=\coverW,height=\coverH]{cover-wrap}};

  % --- FRONT cover (right panel): title block + author + publisher ----
  % Bruno Ace is a wide display face (Regular only), so no bold/italic.
  \node[anchor=center, align=center, text width=6.2in]
    at ($(O)+({\Bn+7+\Sn+3.5},6.7)$){%
      \coverfont\color{white}%
      \contourlength{1.2pt}\contournumber{32}%
      {\LARGE\contour{black}{\booktitle}}\\[16pt]
      {\large\contourlength{0.7pt}\contour{black}{\booksubtitle}}};
  \node[anchor=center, align=center, text width=6.2in]
    at ($(O)+({\Bn+7+\Sn+3.5},1.7)$){%
      \coverfont\color{white}\contourlength{0.7pt}%
      {\large\contour{black}{\bookauthor}}\\[10pt]
      {\normalsize\contour{black}{\bookpublisher}}};

  % --- SPINE (centre panel): title + author, rotated, if thick enough -
  \ifspinetext
    \node[rotate=-90, anchor=center, text=white]
      at ($(O)+({\Bn+7+\Sn/2},5.125)$){%
        \coverfont\small\contourlength{0.5pt}%
        \contour{black}{\booktitle\quad\textbullet\quad\bookauthor}};
  \fi

  % --- BACK cover (left panel): synopsis ------------------------------
  % \contour cannot break paragraphs, so the wrapping blurb sits on a
  % translucent dark panel for legibility over the busy artwork.
  \node[anchor=center, align=flush left, text width=5.3in, text=white,
        fill=black, fill opacity=0.42, text opacity=1,
        rounded corners=8pt, inner sep=16pt]
    at ($(O)+({\Bn+3.5},5.3)$){\large\bookblurb};
  \node[anchor=south, align=center, text=white]
    at ($(O)+({\Bn+3.5},1.0)$){%
      \coverfont\small\contourlength{0.6pt}%
      \contour{black}{gdl.fractalmanifold.com}};
\end{tikzpicture}%
\end{document}
